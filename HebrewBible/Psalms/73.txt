תְּהִלִּים.
73.
מִזְמֹ֗ור לְאָ֫סָ֥ף אַ֤ךְ טֹ֭וב לְיִשְׂרָאֵ֥ל אֱלֹהִ֗ים לְבָרֵ֥י לֵבָֽב׃ 
וַאֲנִ֗י כִּ֭מְעַט *נָטוּי (נָטָ֣יוּ) רַגְלָ֑י כְּ֝אַ֗יִן *שֻׁפְּכָה (שֻׁפְּכ֥וּ) אֲשֻׁרָֽי׃ 
כִּֽי־קִ֭נֵּאתִי בַּֽהֹולְלִ֑ים שְׁלֹ֖ום רְשָׁעִ֣ים אֶרְאֶֽה׃ 
כִּ֤י אֵ֖ין חַרְצֻבֹּ֥ות לְמֹותָ֗ם וּבָרִ֥יא אוּלָֽם׃ 
בַּעֲמַ֣ל אֱנֹ֣ושׁ אֵינֵ֑מֹו וְעִם־אָ֝דָ֗ם לֹ֣א יְנֻגָּֽעוּ׃ 
לָ֭כֵן עֲנָקַ֣תְמֹו גַאֲוָ֑ה יַעֲטָף־שִׁ֝֗ית חָמָ֥ס לָֽמֹו׃ 
יָ֭צָא מֵחֵ֣לֶב עֵינֵ֑מֹו עָ֝בְר֗וּ מַשְׂכִּיֹּ֥ות לֵבָֽב׃ 
יָמִ֤יקוּ ׀ וִידַבְּר֣וּ בְרָ֣ע עֹ֑שֶׁק מִמָּרֹ֥ום יְדַבֵּֽרוּ׃ 
שַׁתּ֣וּ בַשָּׁמַ֣יִם פִּיהֶ֑ם וּ֝לְשֹׁונָ֗ם תִּֽהֲלַ֥ךְ בָּאָֽרֶץ׃ 
לָכֵ֤ן ׀ *יָשִׁיב (יָשׁ֣וּב) עַמֹּ֣ו הֲלֹ֑ם וּמֵ֥י מָ֝לֵ֗א יִמָּ֥צוּ לָֽמֹו׃ 
וְֽאָמְר֗וּ אֵיכָ֥ה יָדַֽע־אֵ֑ל וְיֵ֖שׁ דֵּעָ֣ה בְעֶלְיֹֽון׃ 
הִנֵּה־אֵ֥לֶּה רְשָׁעִ֑ים וְשַׁלְוֵ֥י עֹ֝ולָ֗ם הִשְׂגּוּ־חָֽיִל׃ 
אַךְ־רִ֭יק זִכִּ֣יתִי לְבָבִ֑י וָאֶרְחַ֖ץ בְּנִקָּיֹ֣ון כַּפָּֽי׃ 
וָאֱהִ֣י נָ֭גוּעַ כָּל־הַיֹּ֑ום וְ֝תֹוכַחְתִּ֗י לַבְּקָרִֽים׃ 
אִם־אָ֭מַרְתִּי אֲסַפְּרָ֥ה כְמֹ֑ו הִנֵּ֤ה דֹ֭ור בָּנֶ֣יךָ בָגָֽדְתִּי׃ 
וָֽ֭אֲחַשְּׁבָה לָדַ֣עַת זֹ֑את עָמָ֖ל *הִיא (ה֣וּא) בְעֵינָֽי׃ 
עַד־אָ֭בֹוא אֶל־מִקְדְּשֵׁי־אֵ֑ל אָ֝בִ֗ינָה לְאַחֲרִיתָֽם׃ 
אַ֣ךְ בַּ֭חֲלָקֹות תָּשִׁ֣ית לָ֑מֹו הִ֝פַּלְתָּ֗ם לְמַשּׁוּאֹֽות׃ 
אֵ֤יךְ הָי֣וּ לְשַׁמָּ֣ה כְרָ֑גַע סָ֥פוּ תַ֝֗מּוּ מִן־בַּלָּהֹֽות׃ 
כַּחֲלֹ֥ום מֵהָקִ֑יץ אֲ֝דֹנָי בָּעִ֤יר ׀ צַלְמָ֬ם תִּבְזֶֽה׃ 
כִּ֭י יִתְחַמֵּ֣ץ לְבָבִ֑י וְ֝כִלְיֹותַ֗י אֶשְׁתֹּונָֽן׃ 
וַאֲנִי־בַ֭עַר וְלֹ֣א אֵדָ֑ע בְּ֝הֵמֹ֗ות הָיִ֥יתִי עִמָּֽךְ׃ 
וַאֲנִ֣י תָמִ֣יד עִמָּ֑ךְ אָ֝חַ֗זְתָּ בְּיַד־יְמִינִֽי׃ 
בַּעֲצָתְךָ֥ תַנְחֵ֑נִי וְ֝אַחַ֗ר כָּבֹ֥וד תִּקָּחֵֽנִי׃ 
מִי־לִ֥י בַשָּׁמָ֑יִם וְ֝עִמְּךָ֗ לֹא־חָפַ֥צְתִּי בָאָֽרֶץ׃ 
כָּלָ֥ה שְׁאֵרִ֗י וּלְבָ֫בִ֥י צוּר־לְבָבִ֥י וְחֶלְקִ֗י אֱלֹהִ֥ים לְעֹולָֽם׃ 
כִּֽי־הִנֵּ֣ה רְחֵקֶ֣יךָ יֹאבֵ֑דוּ הִ֝צְמַ֗תָּה כָּל־זֹונֶ֥ה מִמֶּֽךָּ׃ 
וַאֲנִ֤י ׀ קִֽרֲבַ֥ת אֱלֹהִ֗ים לִ֫י־טֹ֥וב שַׁתִּ֤י ׀ בַּאדֹנָ֣י יְהֹוִ֣ה מַחְסִ֑י לְ֝סַפֵּ֗ר כָּל־מַלְאֲכֹותֶֽיךָ׃
