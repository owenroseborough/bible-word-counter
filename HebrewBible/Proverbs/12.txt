מִשְׁלֵי.
12.
אֹהֵ֣ב מ֖וּסָר אֹ֣הֵֽב דָּ֑עַת וְשֹׂנֵ֖א תֹוכַ֣חַת בָּֽעַר׃ 
טֹ֗וב יָפִ֣יק רָ֭צֹון מֵיְהוָ֑ה וְאִ֖ישׁ מְזִמֹּ֣ות יַרְשִֽׁיעַ׃ 
לֹא־יִכֹּ֣ון אָדָ֣ם בְּרֶ֑שַׁע וְשֹׁ֥רֶשׁ צַ֝דִּיקִ֗ים בַּל־יִמֹּֽוט׃ 
אֵֽשֶׁת־חַ֭יִל עֲטֶ֣רֶת בַּעְלָ֑הּ וּכְרָקָ֖ב בְּעַצְמֹותָ֣יו מְבִישָֽׁה׃ 
מַחְשְׁבֹ֣ות צַדִּיקִ֣ים מִשְׁפָּ֑ט תַּחְבֻּלֹ֖ות רְשָׁעִ֣ים מִרְמָֽה׃ 
דִּבְרֵ֣י רְשָׁעִ֣ים אֱרָב־דָּ֑ם וּפִ֥י יְ֝שָׁרִ֗ים יַצִּילֵֽם׃ 
הָפֹ֣וךְ רְשָׁעִ֣ים וְאֵינָ֑ם וּבֵ֖ית צַדִּיקִ֣ים יַעֲמֹֽד׃ 
לְֽפִי־שִׂ֭כְלֹו יְהֻלַּל־אִ֑ישׁ וְנַעֲוֵה־לֵ֝֗ב יִהְיֶ֥ה לָבֽוּז׃ 
טֹ֣וב נִ֭קְלֶה וְעֶ֣בֶד לֹ֑ו מִ֝מְּתַכַּבֵּ֗ד וַחֲסַר־לָֽחֶם׃ 
יֹודֵ֣עַ צַ֭דִּיק נֶ֣פֶשׁ בְּהֶמְתֹּ֑ו וְֽרַחֲמֵ֥י רְ֝שָׁעִ֗ים אַכְזָרִֽי׃ 
עֹבֵ֣ד אַ֭דְמָתֹו יִֽשְׂבַּֽע־לָ֑חֶם וּמְרַדֵּ֖ף רֵיקִ֣ים חֲסַר־לֵֽב׃ 
חָמַ֣ד רָ֭שָׁע מְצֹ֣וד רָעִ֑ים וְשֹׁ֖רֶשׁ צַדִּיקִ֣ים יִתֵּֽן׃ 
בְּפֶ֣שַׁע שְׂ֭פָתַיִם מֹוקֵ֣שׁ רָ֑ע וַיֵּצֵ֖א מִצָּרָ֣ה צַדִּֽיק׃ 
מִפְּרִ֣י פִי־אִ֭ישׁ יִשְׂבַּע־טֹ֑וב וּגְמ֥וּל יְדֵי־אָ֝דָ֗ם *יָשׁוּב (יָשִׁ֥יב) לֹֽו׃ 
דֶּ֣רֶךְ אֱ֭וִיל יָשָׁ֣ר בְּעֵינָ֑יו וְשֹׁמֵ֖עַ לְעֵצָ֣ה חָכָֽם׃ 
אֱוִ֗יל בַּ֭יֹּום יִוָּדַ֣ע כַּעְסֹ֑ו וְכֹסֶ֖ה קָלֹ֣ון עָרֽוּם׃ 
יָפִ֣יחַ אֱ֭מוּנָה יַגִּ֣יד צֶ֑דֶק וְעֵ֖ד שְׁקָרִ֣ים מִרְמָֽה׃ 
יֵ֣שׁ בֹּ֖וטֶה כְּמַדְקְרֹ֣ות חָ֑רֶב וּלְשֹׁ֖ון חֲכָמִ֣ים מַרְפֵּֽא׃ 
שְֽׂפַת־אֱ֭מֶת תִּכֹּ֣ון לָעַ֑ד וְעַד־אַ֝רְגִּ֗יעָה לְשֹׁ֣ון שָֽׁקֶר׃ 
מִ֭רְמָה בְּלֶב־חֹ֣רְשֵׁי רָ֑ע וּֽלְיֹעֲצֵ֖י שָׁלֹ֣ום שִׂמְחָֽה׃ 
לֹא־יְאֻנֶּ֣ה לַצַּדִּ֣יק כָּל־אָ֑וֶן וּ֝רְשָׁעִ֗ים מָ֣לְאוּ רָֽע׃ 
תֹּועֲבַ֣ת יְ֭הוָה שִׂפְתֵי־שָׁ֑קֶר וְעֹשֵׂ֖י אֱמוּנָ֣ה רְצֹונֹֽו׃ 
אָדָ֣ם עָ֭רוּם כֹּ֣סֶה דָּ֑עַת וְלֵ֥ב כְּ֝סִילִ֗ים יִקְרָ֥א אִוֶּֽלֶת׃ 
יַד־חָרוּצִ֥ים תִּמְשֹׁ֑ול וּ֝רְמִיָּ֗ה תִּהְיֶ֥ה לָמַֽס׃ 
דְּאָגָ֣ה בְלֶב־אִ֣ישׁ יַשְׁחֶ֑נָּה וְדָבָ֖ר טֹ֣וב יְשַׂמְּחֶֽנָּה׃ 
יָתֵ֣ר מֵרֵעֵ֣הוּ צַדִּ֑יק וְדֶ֖רֶךְ רְשָׁעִ֣ים תַּתְעֵֽם׃ 
לֹא־יַחֲרֹ֣ךְ רְמִיָּ֣ה צֵידֹ֑ו וְהֹון־אָדָ֖ם יָקָ֣ר חָרֽוּץ׃ 
בְּאֹֽרַח־צְדָקָ֥ה חַיִּ֑ים וְדֶ֖רֶךְ נְתִיבָ֣ה אַל־מָֽוֶת׃
